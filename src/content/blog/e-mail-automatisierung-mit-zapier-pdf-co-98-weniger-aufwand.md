// Regex-Muster für strukturierte Datenextraktion
const extractFormData = (emailBody) => {
    const patterns = {
        name: /Name:\\s*(.+)/i,
        email: /E-Mail:\\s*([^\\s@]+@[^\\s@]+\\.[^\\s@]+)/i,
        company: /Unternehmen:\\s*(.+)/i,
        phone: /Telefon:\\s*(.+)/i,
        service: /Service:\\s*(.+)/i,
        budget: /Budget:\\s*(.+)/i
    };
    
    let extractedData = {};
    
    for (let field in patterns) {
        const match = emailBody.match(patterns[field]);
        extractedData[field] = match ? match[1].trim() : '';
    }
    
    return extractedData;
};

// Validierung der extrahierten Daten
const validateData = (data) => {
    const required = ['name', 'email', 'service'];
    return required.every(field => data[field] && data[field].length > 0);
